dismissal_rule_world_tension = {
	type = dismissal_rules
	ai_will_do = {
		base = 0
		modifier = {  
			add = 1
			OR = {
				has_government = libertarianist
				has_government = centrist
				has_government = liberalist
				has_government = progressivist
				has_government = conservatist
			}
			is_faction_leader = yes 
		}
	}
	dismiss_member_trigger = {
		has_added_tension_amount > 0.1
	}
}
